<forest>

<forest> provides for groups of rooted trees. [19.3 Another Tree Notation]

Example
<forest n="ex5" type="derivation-syntactic">
 <eTree n="Stage 1" xml:id="s1SBAR">
  <label>S'</label>
  <eTree xml:id="s1S">
   <label>S</label>
   <eTree xml:id="s1NP1">
    <label>NP</label>
    <eLeaf>
     <label>you</label>
    </eLeaf>
   </eTree>
   <eTree xml:id="s1VP">
    <label>VP</label>
    <eTree xml:id="s1V">
     <label>V</label>
     <eLeaf>
      <label>do</label>
     </eLeaf>
    </eTree>
   </eTree>
  </eTree>
 </eTree>
 <eTree n="Stage 2" xml:id="s2SBAR" corresp="#s1SBAR">
  <label>S'</label>
  <eTree xml:id="s2S" corresp="#s1S">
   <label>S</label>
   <eTree xml:id="s2NP1" copyOf="#s1NP1">
    <label>NP</label>
   </eTree>
   <eTree xml:id="s2VP" corresp="#s1VP">
    <label>VP</label>
    <eTree xml:id="s2V" copyOf="#s1V">
     <label>V</label>
    </eTree>
    <eTree xml:id="s2NP2" corresp="#s1NP2">
     <label>NP</label>
     <eLeaf corresp="#s1WH">
      <label>t</label>
     </eLeaf>
    </eTree>
   </eTree>
  </eTree>
 </eTree>
</forest>
