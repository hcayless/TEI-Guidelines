<performance>

<performance> contains a section of front or back matter describing how a dramatic piece is to be performed in general or how it was performed on some specific occasion. [7.1.3 Records of Performances 7.1 Front and Back Matter ]
Moduledrama — Performance Texts

Declaration

<rng:element name="performance">
 <rng:ref name="att.global.attributes"/>
 <rng:ref name="att.global.linking.attributes"/>
 <rng:ref name="att.global.analytic.attributes"/>
 <rng:ref name="att.global.facs.attributes"/>
 <rng:ref name="att.global.change.attributes"/>
 <rng:group>
  <rng:zeroOrMore>
   <rng:choice>
    <rng:ref name="model.divTop"/>
    <rng:ref name="model.global"/>
   </rng:choice>
  </rng:zeroOrMore>
  <rng:oneOrMore>
   <rng:group>
    <rng:ref name="model.common"/>
   </rng:group>
   <rng:zeroOrMore>
    <rng:ref name="model.global"/>
   </rng:zeroOrMore>
  </rng:oneOrMore>
  <rng:zeroOrMore>
   <rng:group>
    <rng:ref name="model.divBottom"/>
   </rng:group>
   <rng:zeroOrMore>
    <rng:ref name="model.global"/>
   </rng:zeroOrMore>
  </rng:zeroOrMore>
 </rng:group>
</rng:element>
element performance
{
   att.global.attributes,
   att.global.linking.attributes,
   att.global.analytic.attributes,
   att.global.facs.attributes,
   att.global.change.attributes,
   (
      ( model.divTop | model.global )*,
      ( ( model.common ), model.global* )+,
      ( ( model.divBottom ), model.global* )*
   )
}
